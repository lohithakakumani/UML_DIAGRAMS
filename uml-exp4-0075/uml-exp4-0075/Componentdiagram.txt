C4Component
title Component diagram for Traffic Management System

Container_Boundary(trafficControl, "Traffic Control System") {
    Component(controlSystem, "Traffic Control System", "Main System Component", "Monitors traffic, manages signals, and detects violations.")
    Component(signalSystem, "Traffic Signal System", "Connected to Traffic Control System", "Changes signal states and syncs with control system.")
    Component(vehicleMonitoring, "Vehicle Monitoring System", "Monitors Vehicles & Violations", "Detects speed and registers violations.")
    Component(cameraSystem, "Traffic Camera System", "Captures Violations", "Captures images and sends data to control system.")
    Component(policeSystem, "Police System", "Handles Traffic Violations", "Issues tickets and receives violation data.")
    Component(emergencyVehicle, "Emergency Vehicle System", "Manages Priority Vehicles", "Requests signal overrides and updates priority status.")
    Component(pedestrianCrossing, "Pedestrian Crossing System", "Manages Pedestrian Signals", "Requests crossing and activates signals.")

    Rel(controlSystem, signalSystem, "Uses", "monitorTraffic(), manageTrafficSignals(), detectViolations()")
    Rel(signalSystem, controlSystem, "Uses", "changeSignalState(), syncWithControlSystem()")
    Rel(vehicleMonitoring, cameraSystem, "Depends on", "detectSpeed(), registerViolation()")
    Rel(cameraSystem, controlSystem, "Sends data to", "captureImage(), sendDataToControlSystem()")
    Rel(policeSystem, cameraSystem, "Depends on", "issueTicket(), receiveViolationData()")
    Rel(policeSystem, controlSystem, "Depends on", "issueTicket(), receiveViolationData()")
    Rel(emergencyVehicle, controlSystem, "Depends on", "requestSignalOverride(), updatePriorityStatus()")
    Rel(pedestrianCrossing, controlSystem, "Depends on", "requestCrossing(), activateSignal()")
}
