title Deployment Diagram for Traffic Management System

Deployment_Node(tc_server, "Traffic Control Server", "Central System Processing, Linux") {
    Container(traffic_mgmt_software, "Traffic Management Software", "Java", "Manages traffic flow and signals.")
    ContainerDb(database_system, "Database System", "Relational Database", "Stores traffic data and logs.")
}

Deployment_Node(signal_controller, "Traffic Signal Controller", "Embedded RTOS") {
    Container(signal_control_module, "Signal Control Module", "C/C++", "Controls traffic lights at intersections.")
    Container(network_interface, "Network Interface", "Embedded Software", "Handles communication with the central system.")
}

Deployment_Node(camera_system, "Traffic Camera System", "Linux") {
    Container(camera_processing_software, "Camera Processing Software", "Python", "Processes video feeds from traffic cameras.")
    Container(ai_violation_detection, "AI Violation Detection", "Machine Learning", "Detects traffic violations using AI.")
}

Deployment_Node(police_monitoring, "Police Monitoring System", "Windows") {
    Container(violation_db_access, "Violation Database Access", "C#", "Accesses violation records.")
    Container(ticket_issuing_system, "Ticket Issuing System", "Java", "Issues tickets for violations.")
}

Deployment_Node(emergency_vehicle, "Emergency Vehicle System", "Android") {
    Container(emergency_signal_override, "Emergency Signal Override", "Java", "Overrides traffic signals for emergency vehicles.")
    Container(vehicle_tracking, "Vehicle Tracking", "Java", "Tracks emergency vehicle locations.")
}

Deployment_Node(user_app, "User Web & Mobile App", "iOS & Android") {
    Container(traffic_updates, "Traffic Updates", "JavaScript", "Provides real-time traffic updates.")
    Container(violation_reports, "Violation Reports", "JavaScript", "Displays traffic violation reports.")
    Container(emergency_requests, "Emergency Requests", "JavaScript", "Allows users to send emergency requests.")
}

Rel(traffic_mgmt_software, database_system, "Reads from and writes to", "HTTPS")
Rel(signal_control_module, network_interface, "Communicates with", "MQTT")
Rel(camera_processing_software, ai_violation_detection, "Processes data for", "TCP/IP")
Rel(violation_db_access, ticket_issuing_system, "Accesses data from", "HTTPS")
Rel(emergency_signal_override, vehicle_tracking, "Sends signals to", "4G/LTE")
Rel(traffic_updates, violation_reports, "Provides updates to", "HTTPS")
